58template<std::size_t K>

60{
61 std::array<double, K> x;
62 for (auto i = 0u; i < K; ++i) { x[i] = -detail::cos_s(2. * pi_v<double> * i / (2 * K - 1)); }

87template<std::size_t K, std::size_t I = 8>
88 requires(K <= 40)
89constexpr std::pair<std::array<double, K>, std::array<double, K>> lgr_nodes()
90{
91 // initial guess
92 auto xs = cgr_nodes<K>();
93
94 // initialize weights
95 std::array<double, K> ws;
96 ws[0] = 2. / (K * K);
97
98 // two rightmost column of legendre basis matrix
99 constexpr auto B = polynomial_basis<PolynomialBasis::Legendre, K>().template block<K + 1, 2>(0, K - 1);
100
101 for (auto i = 1u; i < K; ++i) {
102 StaticMatrix<double, 1, 2> U_B;
103 for (auto iter = 0u; iter < I; ++iter) {
104 U_B = monomial_derivative<K>(xs[i], 0) * B;
105 const auto dU_B = monomial_derivative<K>(xs[i], 1) * B;
106 const double fxi = U_B[0][0] + U_B[0][1];
107 const double dfxi = dU_B[0][0] + dU_B[0][1];
108 xs[i] -= fxi / dfxi;
109 }
110 ws[i] = (1. - xs[i]) / (K * K * U_B[0][0] * U_B[0][0]);
111 }
112
113 return {xs, ws};
114}
