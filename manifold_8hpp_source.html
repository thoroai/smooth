31template<typename M>
32concept Manifold =
33requires (Eigen::Index dof) {
34 // Underlying scalar type
35 typename traits::man<M>::Scalar;
36 // Default representation
37 typename traits::man<M>::PlainObject;
38 // Compile-time degrees of freedom (tangent space dimension). Can be dynamic (equal to -1)
39 {traits::man<M>::Dof}->std::convertible_to<int>;
40 // A default-initialized Manifold object (if Dof > 0 then dof = Dof can be assumed)
41 {traits::man<M>::Default(dof)}->std::convertible_to<typename traits::man<M>::PlainObject>;
42} &&
43requires(const M & m1, const M & m2, const Eigen::Vector<typename traits::man<M>::Scalar, traits::man<M>::Dof> & a) {
44 // Run-time degrees of freedom (tangent space dimension)
45 {traits::man<M>::dof(m1)}->std::convertible_to<Eigen::Index>;
46 // Right-plus: add a tangent vector to a Manifold element to obtain a new Manifold element
47 {traits::man<M>::rplus(m1, a)}->std::convertible_to<typename traits::man<M>::PlainObject>;
48 // Right-minus: subtract a Manifold element from another to obtain the difference tangent vector
49 {traits::man<M>::rminus(m1, m2)}->std::convertible_to<Eigen::Vector<typename traits::man<M>::Scalar, traits::man<M>::Dof>>;
50} && (
51 !std::is_convertible_v<typename traits::man<M>::Scalar, double> ||
52 requires (const M & m) {
53 {traits::man<M>::template cast<double>(m)}->std::convertible_to<typename traits::man<M>::template CastT<double>>;
54 }
55) && (
56 !std::is_convertible_v<typename traits::man<M>::Scalar, float> ||
57 requires (const M & m) {
58 {traits::man<M>::template cast<float>(m)}->std::convertible_to<typename traits::man<M>::template CastT<float>>;
59 }
60) &&
61// PlainObject must be default-constructible
62std::is_default_constructible_v<typename traits::man<M>::PlainObject> &&
63std::is_copy_constructible_v<typename traits::man<M>::PlainObject> &&
64// PlainObject must be assignable from M
65std::is_assignable_v<typename traits::man<M>::PlainObject &, M>;
