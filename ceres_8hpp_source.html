73auto dr_ceres(auto && f, auto && x)
74{
75 // There is potential to improve thie speed of this by reducing casting.
76 // The ceres Jet type supports binary operations with e.g. double, but currently
77 // the Lie operations require everything to have a uniform scalar type. Enabling
78 // plus and minus for different scalars would thus save some casts.
79 using Result = decltype(std::apply(f, x));
80 using Scalar = ::smooth::Scalar<Result>;
81
82 static_assert(Manifold<Result>, "f(x) is not a Manifold");
83
84 Result fval = std::apply(f, x);
85
86 static constexpr Eigen::Index Nx = wrt_Dof<decltype(x)>();
87 static constexpr Eigen::Index Ny = Dof<Result>;
88 const Eigen::Index nx = std::apply([](auto &&... args) { return (dof(args) + ...); }, x);
89 const Eigen::Index ny = dof<Result>(fval);
90
91 static_assert(Nx > 0, "Ceres autodiff does not support dynamic sizes");
92
93 Eigen::Matrix<Scalar, Nx, 1> a = Eigen::Matrix<Scalar, Nx, 1>::Zero(nx);
94 Eigen::Matrix<Scalar, Ny, 1> b(ny);
95 Eigen::Matrix<Scalar, Ny, Nx, (Nx == 1) ? Eigen::ColMajor : Eigen::RowMajor> jac(ny, nx);
96 jac.setZero();
97
98 const auto f_deriv = [&]<typename T>(const T * in, T * out) {
99 Eigen::Map<const Eigen::Matrix<T, Nx, 1>> mi(in, nx);
100 Eigen::Map<Eigen::Matrix<T, Ny, 1>> mo(out, ny);
101 mo = rminus<CastT<T, Result>>(std::apply(f, wrt_rplus(wrt_cast<T>(x), mi)), cast<T, Result>(fval));
102 return true;
103 };
104 const Scalar * a_ptr[1] = {a.data()};
105 Scalar * jac_ptr[1] = {jac.data()};
106
107 ceres::internal::AutoDifferentiate<Dof<Result>, ceres::internal::StaticParameterDims<Nx>>(
108 f_deriv, a_ptr, static_cast<int>(b.size()), b.data(), jac_ptr);
109
110 return std::make_pair(std::move(fval), Eigen::Matrix<Scalar, Ny, Nx>(jac));
111}
