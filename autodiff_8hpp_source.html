44template<std::size_t K = 1>
45 requires(K >= 1 && K <= 2)
46auto dr_autodiff(auto && f, auto && x)
47{
48 using Result = decltype(std::apply(f, x));
49 using Scalar = ::smooth::Scalar<Result>;
50 using Eigen::Matrix;
51
52 static_assert(Manifold<Result>, "f(x) is not a Manifold");
53
54 using AdScalar = autodiff::HigherOrderDual<K, Scalar>;
55
56 Result F = std::apply(f, x);
57
58 static constexpr auto Nx = wrt_Dof<decltype(x)>();
59 static constexpr auto Ny = Dof<Result>;
60 const Eigen::Index nx = std::apply([](auto &&... args) { return (dof(args) + ...); }, x);
61 const Eigen::Index ny = dof(F);
62
63 // cast F and x to ad types
64 const auto x_ad = wrt_cast<AdScalar>(x);
65 CastT<AdScalar, Result> F_ad = cast<AdScalar>(F);
66
67 if constexpr (K == 1) {
68 // zero-valued tangent element
69 Matrix<AdScalar, Nx, 1> a_ad = Matrix<AdScalar, Nx, 1>::Zero(nx);
70
71 // function to differentiate
72 const auto f_ad = [&](Matrix<AdScalar, Nx, 1> & var) -> Matrix<AdScalar, Ny, 1> {
73 return rminus<CastT<AdScalar, Result>>(std::apply(f, wrt_rplus(x_ad, var)), F_ad);
74 };
75
76 Matrix<Scalar, Ny, Nx> J(ny, nx);
77
78 J = autodiff::jacobian(f_ad, autodiff::wrt(a_ad), autodiff::at(a_ad));
79 return std::make_pair(std::move(F), std::move(J));
80 } else if constexpr (K == 2) {
81 // function to differentiate
82 const auto f_ad =
83 [&f, &x_ad, &F_ad](Matrix<AdScalar, Nx, 1> & var1, Matrix<AdScalar, Nx, 1> & var2) -> Matrix<AdScalar, Ny, 1> {
84 return rminus(std::apply(f, wrt_rplus(wrt_rplus(x_ad, var1), var2)), F_ad);
85 };
86
87 Matrix<Scalar, Ny, Nx> J(ny, nx);
88 Matrix<Scalar, Nx, std::min(Nx, Ny) == -1 ? -1 : Nx * Ny> H(nx, nx * ny);
89
90 // zero-valued tangent elements
91 Matrix<AdScalar, Nx, 1> a_ad1 = Matrix<AdScalar, Nx, 1>::Zero(nx);
92 Matrix<AdScalar, Nx, 1> a_ad2 = Matrix<AdScalar, Nx, 1>::Zero(nx);
93
94 const auto a_wrt1 = autodiff::wrt(a_ad1);
95 const auto a_wrt2 = autodiff::wrt(a_ad2);
96
97 autodiff::detail::ForEachWrtVar(a_wrt1, [&](auto && i, auto && xi) constexpr {
98 autodiff::detail::ForEachWrtVar(a_wrt2, [&](auto && j, auto && xj) constexpr {
99 const auto u = autodiff::detail::eval(f_ad, autodiff::at(a_ad1, a_ad2), autodiff::wrt(xi, xj));
100 for (auto k = 0u; k < ny; ++k) {
101 J(k, i) = static_cast<double>(autodiff::detail::derivative<1>(u[k]));
102 H(j, k * nx + i) = autodiff::detail::derivative<2>(u[k]);
103 }
104 });
105 });
106
107 return std::make_tuple(std::move(F), std::move(J), std::move(H));
108 }
109}
